38void exhaustive(OutputData &output, Algorithm &data) {
39 int n = data.pallets.size();
40 int maxProfit = 0;
41 vector<Pallet> bestCombination;
42 vector<bool> usedItems(n, false);
43
44 bool done = false;
45 while (!done) {
46
47 int weight = 0, profit = 0;
48 vector<Pallet> current;
49
50 for (int i = 0; i < n; ++i) {
51 if (usedItems[i]) {
52 weight += data.pallets[i].weight;
53 profit += data.pallets[i].profit;
54 if (weight > data.TotalWeight) break;
55 current.push_back(data.pallets[i]);
56 }
57 }
58
59 if (weight <= data.TotalWeight && profit > maxProfit) {
60 maxProfit = profit;
61 bestCombination = current;
62 }
63
64 int i = 0;
65 while (i < n) {
66 if (!usedItems[i]) {
67 usedItems[i] = true;
68 break;
69 } else {
70 usedItems[i] = false;
71 ++i;
72 }
73 }
74 if (i == n) done = true;
75 }
76
77 output.totalProfit = maxProfit;
78 output.pallets = bestCombination;

97void knapsackDP(OutputData &output, const Algorithm &data) {
98 const int maxWeight = data.TotalWeight;
99 const int n = data.pallets.size();
100
101 vector<vector<int>> dp(n + 1, vector<int>(maxWeight + 1, 0));
102
103 for (int i = 1; i <= n; ++i) {
104 int w = data.pallets[i - 1].weight;
105 int p = data.pallets[i - 1].profit;
106
107 for (int j = 0; j <= maxWeight; ++j) {
108 if (w <= j) {
109 dp[i][j] = max(dp[i - 1][j], dp[i - 1][j - w] + p);
110 } else {
111 dp[i][j] = dp[i - 1][j];
112 }
113 }
114 }
115
116 int j = maxWeight;
117 for (int i = n; i > 0; --i) {
118 if (dp[i][j] != dp[i - 1][j]) {
119 const Pallet &p = data.pallets[i - 1];
120 output.pallets.push_back(p);
121 output.totalProfit += p.profit;
122 j -= p.weight;
123 }
124 }
125
126 reverse(output.pallets.begin(), output.pallets.end());
127}

259 int profit = 0, weight = 0;

338void knapsackGA(OutputData &output, const Algorithm &data) {
339 if (data.pallets.empty() || data.TotalWeight <= 0) return;
340
341 const int maxWeight = data.TotalWeight;
342 const int n = data.pallets.size();
343
344 // These values can be changed
345 const int generations = 200; // Number of cycles
346 const int populationSize = 50; // Number of possible solutions in each generation
347 const double crossoverRate = 0.8; // Chance that 2 solutions will "mix" into a new solution
348 const double mutationRate = 0.05; // Chance that a bit in a solution will randomly flip (from 0 to 1 and vice versa)
349
350 if (populationSize < 2 || n == 0) return;
351
352 // RNG setup
353 random_device rd; // Create a source of random data
354 mt19937 rng(rd()); // Number Generator
355 uniform_real_distribution<double> probDist(0.0, 1.0); // Random number between 0 and 1
356 uniform_int_distribution<int> populationIndexDist(0, populationSize - 1); // Random number between 0 and the population size - 1 (Random population index)
357 uniform_int_distribution<int> geneIndexDist(0, n - 1); // Random number between 0 and the number of genes (pallets) - 1 (Random pallet index in a candidate solution)
358
359 using Chromosome = vector<bool>; // Each Chromosome (Candidate solution) is a vector that represents which pallets are included
360
361 // Make the initial population
362 vector<Chromosome> population(populationSize);
363 for (auto& chrom : population) {
364 chrom = generateRandomChromosome(n, maxWeight, data, rng);
365 }
366
367 // Track the best solution
368 Chromosome bestChrom(n, false);
369 int bestFitness = 0;
370
371 for (int gen = 0; gen < generations; ++gen) {
372 vector<pair<int, Chromosome>> evaluated;
373 for (const auto &chrom : population) {
374 int f = fitnessFunction(chrom, data, maxWeight);
375 evaluated.emplace_back(f, chrom);
376
377 if (f > bestFitness) {
378 // Save the best solution so far
379 bestFitness = f;
380 bestChrom = chrom;
381 }
382 }
383
384 sort(evaluated.begin(), evaluated.end(), greater<>());
385
386 if (evaluated.size() < 2) continue;
387
388 // Keep the 2 best solutions to be guaranteed to be kept on the next generation, so they're not lost to rng
389 vector<Chromosome> newPopulation = {
390 evaluated[0].second,
391 evaluated[1].second
392 };
393
394 // Create a new generation
395 while (newPopulation.size() < populationSize) {
396 // Randomly pick 2 parents
397 const Chromosome &parent1 = evaluated[populationIndexDist(rng)].second;
398 const Chromosome &parent2 = evaluated[populationIndexDist(rng)].second;
399
400 // Start with the child equal to one of the parents
401 Chromosome child = parent1;
402
403 // There's a chance that the child will be a combination of the 2 parents,
404 // keeping the first 'point' (randomly generated value) pallets of the first, and the rest of the pallets be the same as the other parent
405 if (probDist(rng) < crossoverRate) {
406 int point = geneIndexDist(rng);
407 for (int i = point; i < n; ++i) {
408 child[i] = parent2[i];
409 }
410 }
411
412 // Apply the mutation chance
413 for (int i = 0; i < n; ++i) {
414 if (probDist(rng) < mutationRate) {
415 child[i] = !child[i];
416 }
417 }
418
419 newPopulation.push_back(child); // Add the new chromosome to the next generation
420 }
421
422 population = newPopulation; // Start treating the new generation
423 }
424
425 // Compute the final solution's value, adding the used pallets to the output
426 for (int i = 0; i < n; ++i) {
427 if (bestChrom[i]) {
428 const Pallet &p = data.pallets[i];
429 output.pallets.push_back(p);
430 output.totalProfit += p.profit;
431 }
432 }
